Hive import by default appends data to existing ones
sqoop import \
--connect jdbc:mysql://quickstart.cloudera/retail_db \
--username root \
--password cloudera \
--table order_items \
--hive-import \
--hive-database hive_retail_db \
--hive-table order_items \
--num-mappers 3

If you want to overwrite existing table uding hive import
sqoop import \
--connect jdbc:mysql://quickstart.cloudera/retail_db \
--username root \
--password cloudera \
--table order_items \
--hive-import \
--hive-table hive_retail_db.order_items \
--hive-overwrite \
--num-mappers 2
